host1
----------
ssh user@host1 -M -S /tmp/host1.sock #Multiplexed

ssh user@host1 -S /tmp/host1.sock #Interactive

#Create tunnel associated with Multiplexed socket /tmp/host1.sock Kali port 2222 going through host to port 22
ssh user@host1 -S /tmp/host1.sock -O forward -L2222:host2:22 # noninteractive
ssh user@host1 -S /tmp/host1.sock -L2222:host2:22 # interactive

ssh user@host -S /tmp/host1.sock -O forward -R32123:127.0.0.1:4444 # noninteractive REMOTE listener to local nc on 4444
nc -lp 4444
ssh user@host -S /tmp/host1.sock -O cancel -R32123:127.0.0.1:4444 # noninteractive CLOSE REMOTE listener


host2
----------
ssh user@127.0.0.1 -p 2222 -M -S /tmp/host2.sock # Multiplexed

ssh user@127.0.0.1 -p 2222 -S /tmp/host2.sock # Interactive

ssh user@127.0.0.1 -p 2222 -S /tmp/host2.sock -L3333:host3:22 # interactive forward tunnel for connecting to next host

-----------
