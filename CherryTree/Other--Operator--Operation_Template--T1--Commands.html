Commands


General:
whoami && hostname && date && date -u && cat /etc/timezone && w && egrep 'sudo|wheel' /etc/group
ls around

Network and Process:
ip a
netstat -natup
ps -elfH
history
cat .bash_history

File Poking:
ls -al /tmp # Look for previous exploits/actors
ls -al /dev/shm # Same as tmp (tmp is often noexec, which /dev/shm is not)
cat /etc/passwd # Any users other than root with uid/gid of 0
ls -latr /var/log # Look for receng logs
last # Last time users were on the box
lastb # Last failed login attempts
ls latr /etc # Look for modified configuration files
ls -al /home/*/.ssh/* # User private ssh keys
ls -al /root/.ssh/* # Root private ssh keys
ls -latr /etc/initd/ # Startup scripts
egrep -h '^\.\ /' /etc/init.d/* | sort -u # Look for source files
egrep -ho '\.\ /[^\ ]+' /etc/init.d/* | sort -u # More aggressive searh
find / -mmin -60 -type f -ls # FInd a file thats been modified last 60 minutes
find / -path /proc -prune -o -path /run -prune -o -path /sys -prune -o -mmin -60 -type f -ls

Cron Jobs:
cat /etc/crontab
crontab -l
crontab -l -u root
ls -al /var/spool/cron/crontabs/
ls -al /etc/cron*

Startup Files:
ls -al /etc/rc.local
more /etc/rc.local
echo $PATH
